from scipy.fftpack import fft2, fftshift
from skimage.transform import warp_polar


imageDir='_images/'

# remember results='asis' for the output to be right
def FilterImagesInFolder(imageDir):
  for imageFileStub in os.listdir(imageDir):
    
    fileWithoutExt=os.path.splitext(imageFileStub)[0]
    print("### "+fileWithoutExt+"\n")
    outpath="procimage/"+fileWithoutExt+"/"
    os.makedirs(outpath, exist_ok = True)
  
    image_gray = cv.imread(imageDir+imageFileStub,cv.IMREAD_ANYDEPTH )
    image_fs = np.abs(fftshift(fft2(image_gray)))
    
    img_gray_255=np.uint8(PercentileScaledImage(image_gray,lowerPerc=0.1,upperPerc=99.9,maxVal=255,invert=False))
    SaveImageAndMarkdown(outpath+fileWithoutExt+ " Input Image.png",255-img_gray_255,"#### Input Image")

    scaleFFT=PercentileScaledImage(image_fs,lowerPerc=0.1,upperPerc=99.9,maxVal=255,invert=False)
    fig=plt.subplots(figsize=(10,10))
    plt.imshow(scaleFFT,cmap='magma')
    SavePlotAndMarkdown(outpath+fileWithoutExt+"FFT.png","#### FFT")
    plt.close()
    
    fig=plt.subplots(figsize=(10,10))
    plt.imshow(scaleFFT[1024-200:1024+200,1024-200:1024+200],cmap='magma')
    SavePlotAndMarkdown(outpath+fileWithoutExt+"subFFT.png","#### FFT centre zoomed")
    plt.close()

    subImage=3
    sqSide=image_gray.shape[0]//subImage
    
    fig, ax = plt.subplots(subImage, subImage, figsize=(10,10),
                           sharex=True, sharey=True)
    ax = ax.ravel()
    
    i=0
    for y in range(subImage):
      for x in range(subImage):
        subImageData=image_gray[y*sqSide:(y+1)*sqSide,sqSide*x:(x+1)*sqSide]
        image_fs = np.abs(fftshift(fft2(subImageData)))
        ax[i].imshow(PercentileScaledImage(subImageData,lowerPerc=0.1,upperPerc=99.9,maxVal=255,invert=False),cmap='gray')
        i += 1

    for a in ax:
      a.axis('off')

    fig.tight_layout()
    SavePlotAndMarkdown(outpath+fileWithoutExt+str(x)+str(y)+"raw.png","#### FFT subimage")
    plt.close()
    
    fig, ax = plt.subplots(subImage, subImage, figsize=(10,10),
                           sharex=True, sharey=True)
    ax = ax.ravel()
    
    i=0
    for y in range(subImage):
      for x in range(subImage):
        subImageData=image_gray[y*sqSide:(y+1)*sqSide,sqSide*x:(x+1)*sqSide]
        image_fs = np.abs(fftshift(fft2(subImageData)))
        scaleFFT=PercentileScaledImage(image_fs,lowerPerc=0.1,upperPerc=99.9,maxVal=255,invert=False)
        ax[i].imshow(scaleFFT,cmap='magma')
        i += 1

    for a in ax:
      a.axis('off')

    fig.tight_layout()
    SavePlotAndMarkdown(outpath+fileWithoutExt+str(x)+str(y)+"FFT.png","#### FFT subimage")
      
    plt.close()

      
    
    
FilterImagesInFolder(imageDir)
